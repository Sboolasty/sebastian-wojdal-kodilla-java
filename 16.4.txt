SELECT i.ID, i.SUMMARY, il.NAME as LIST_NAME
FROM ISSUESS i
INNER JOIN ISSUESLISTS il ON i.ISSUESLIST_ID = il.ID;


SELECT i.ID, i.SUMMARY, u.FIRST_NAME, u.LAST_NAME
FROM ISSUESS i
LEFT JOIN USERS u ON i.USER_ID_ASSIGNEDTO = u.ID;


SELECT u.FIRST_NAME, u.LAST_NAME, COUNT(i.ID) as NUM_OF_ISSUES
FROM ISSUESS i
LEFT JOIN USERS u ON i.USER_ID_ASSIGNEDTO = u.ID
GROUP BY u.ID;
